    <script>
        // --- Three.js Setup (Hearts) ---
        const scene = new THREE.Scene();
        const camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);
        const renderer = new THREE.WebGLRenderer({ alpha: true, antialias: true });
        renderer.setSize(window.innerWidth, window.innerHeight);
        document.getElementById('canvas-container').appendChild(renderer.domElement);

        const hearts = [];
        const heartShape = new THREE.Shape();
        heartShape.moveTo(0, 0);
        heartShape.bezierCurveTo(0, -0.3, -0.6, -0.3, -0.6, 0);
        heartShape.bezierCurveTo(-0.6, 0.3, 0, 0.6, 0, 1);
        heartShape.bezierCurveTo(0, 0.6, 0.6, 0.3, 0.6, 0);
        heartShape.bezierCurveTo(0.6, -0.3, 0, -0.3, 0, 0);

        const extrudeSettings = { depth: 0.2, bevelEnabled: true, bevelSegments: 2, steps: 2, bevelSize: 0.1, bevelThickness: 0.1 };
        const geometry = new THREE.ExtrudeGeometry(heartShape, extrudeSettings);

        for(let i=0; i<25; i++) {
            const material = new THREE.MeshPhongMaterial({ color: 0xff2d75, shininess: 100 });
            const heart = new THREE.Mesh(geometry, material);
            heart.position.set((Math.random()-0.5)*20, (Math.random()-0.5)*20, (Math.random()-0.5)*10);
            heart.rotation.z = Math.PI;
            heart.rotation.y = Math.random() * Math.PI;
            heart.scale.set(0.5, 0.5, 0.5);
            heart.userData = { 
                speed: 0.005 + Math.random()*0.01,
                offset: Math.random() * Math.PI * 2,
                originalY: heart.position.y 
            };
            hearts.push(heart);
            scene.add(heart);
        }

        const light = new THREE.DirectionalLight(0xffffff, 1);
        light.position.set(0, 0, 10);
        scene.add(light);
        scene.add(new THREE.AmbientLight(0xffffff, 0.3));
        camera.position.z = 15;

        function animate() {
            requestAnimationFrame(animate);
            hearts.forEach(h => {
                h.rotation.y += 0.01;
                h.position.y = h.userData.originalY + Math.sin(Date.now() * 0.001 + h.userData.offset) * 0.5;
            });
            renderer.render(scene, camera);
        }
        animate();

        // --- Navigation Logic ---
        let currentStep = 1;

        function nextStep(step) {
            const currentEl = document.getElementById(`step${currentStep}`);
            const nextEl = document.getElementById(`step${step}`);
            
            gsap.to(currentEl, { opacity: 0, scale: 0.8, duration: 0.5, onComplete: () => {
                currentEl.style.display = 'none';
                nextEl.style.display = 'block';
                gsap.fromTo(nextEl, { opacity: 0, scale: 1.2 }, { opacity: 1, scale: 1, duration: 0.8, ease: "back.out(1.7)" });
                document.getElementById('progress-bar').style.width = (step * 20) + '%';
            }});
            currentStep = step;
        }

        // --- Step 4 Polaroid Parallax ---
        const polaroid = document.getElementById('polaroid');
        document.addEventListener('mousemove', (e) => {
            if(currentStep === 4) {
                const xAxis = (window.innerWidth / 2 - e.pageX) / 25;
                const yAxis = (window.innerHeight / 2 - e.pageY) / 25;
                polaroid.style.transform = `rotateY(${xAxis}deg) rotateX(${yAxis}deg) rotateZ(-3deg)`;
            }
        });

        // --- Finale ---
        function startFinale() {
            const btn = document.getElementById('celebrate-btn');
            const msg = document.getElementById('final-message');
            btn.style.pointerEvents = 'none';
            
            gsap.to(btn, { opacity: 0, y: 20, duration: 0.5 });
            setTimeout(() => {
                msg.classList.remove('hidden');
                gsap.from(msg, { opacity: 0, y: 20, duration: 1, ease: "power4.out" });
            }, 600);

            // Confetti Cannon
            const duration = 5 * 1000;
            const end = Date.now() + duration;

            (function frame() {
                confetti({
                    particleCount: 3,
                    angle: 60,
                    spread: 55,
                    origin: { x: 0, y: 0.8 },
                    colors: ['#ff2d75', '#ff4b2b', '#ffffff']
                });
                confetti({
                    particleCount: 3,
                    angle: 120,
                    spread: 55,
                    origin: { x: 1, y: 0.8 },
                    colors: ['#ff2d75', '#ff4b2b', '#ffffff']
                });

                if (Date.now() < end) {
                    requestAnimationFrame(frame);
                }
            }());

            // Emoji Shower
            setTimeout(() => {
                const scalar = 2;
                const heartEmoji = confetti.shapeFromText({ text: '‚ù§Ô∏è', scalar });
                const starEmoji = confetti.shapeFromText({ text: '‚ú®', scalar });

                confetti({
                    shapes: [heartEmoji, starEmoji],
                    particleCount: 40,
                    spread: 100,
                    origin: { y: 0.6 }
                });
            }, 1000);

            // Three.js Hearts Swirl Away
            hearts.forEach((h, i) => {
                gsap.to(h.position, {
                    x: h.position.x * 5,
                    y: 25,
                    z: -20,
                    duration: 4,
                    delay: i * 0.05,
                    ease: "power2.in"
                });
                gsap.to(h.material, { opacity: 0, duration: 4, delay: i * 0.05 });
            });
        }

        // Initial entry
        window.onload = () => {
            gsap.fromTo("#step1", { opacity: 0, y: 50 }, { opacity: 1, y: 0, duration: 1.2, delay: 0.5, display: 'block' });
        };

        window.addEventListener('resize', () => {
            camera.aspect = window.innerWidth / window.innerHeight;
            camera.updateProjectionMatrix();
            renderer.setSize(window.innerWidth, window.innerHeight);
        });
